string intToRoman(int num) {
    string str = "";
    
    map<int, string> change_map{
        {1, "I"},
        {4, "IV"},
        {5, "V"},
    };
    map<int, string>::reverse_iterator iter;
    for (iter = change_map.rbegin(); iter != change_map.rend(); iter++)
    {
        while(num >= iter->first)
        {
            str.append(iter->second);
            num -= iter->first;
        }
    }
