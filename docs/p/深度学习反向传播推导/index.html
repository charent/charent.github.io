import math
class sigmoid:
    def __init__(self):
        self.y = None
    
    def forward(self, x: float) -> float:
        # sigmoid func
        y = 1 / (1 + math.exp( -x ))
        self.y = y

        return y
    
    def backward(self, out: float) -> float:
        dx = out * self.y * (1.0 - self.y)

        return dx
