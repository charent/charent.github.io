class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        n = len(points)

        def get_distance(a: list[int], b: list[int]) -> int:
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        int_max = 2 ** 31
        ans = 0

        graph = [[0 for _ in range(n)] for _ in range(n)]
        for i in range(n):
            for j in range(n):
                dist = get_distance(points[i], points[j])
                graph[i][j] = dist
                graph[j][i] = dist
        
        low_cost = [int_max for _ in range(n)]
        vec = [0 for _ in range(n)]

        vec[0] = 1 # ä»é¡¶ç‚¹0å¼€å§‹
     
        for i in range(1, n):
            low_cost[i] = graph[0][i] # é¡¶ç‚¹0åˆ°içš„è·ç¦»
        
        for _ in range(1, n):
            # æ‰¾å‡ºåˆ©vecä¸­ä¸ºç¦»æ‰€æœ‰1çš„ç‚¹æœ€è¿‘çš„ç‚¹
            min_idx = -1 
            min_cost = int_max

            for j in range(n):
                # vec[j] = 0æ²¡æœ‰åŠ å…¥æœ€å°ç”Ÿæˆæ ‘åˆé›†çš„èŠ‚ç‚¹j
                if vec[j] == 0 and low_cost[j] < min_cost:
                    min_idx = j
                    min_cost = low_cost[j]
            
            vec[min_idx] = 1
            ans += min_cost
          

            # æ›´æ–°vecä¸­çš„æ‰€æœ‰lowcost
            for j in range(n):
                # åˆå§‹åŒ–æ—¶low_cost[j]ä¸º0åˆ°jçš„è·ç¦»ï¼ŒåŠ å…¥æ–°çš„èŠ‚ç‚¹min_idxåï¼Œæœ€å°ç”Ÿæˆæ ‘åˆ°æœªåŠ å…¥èŠ‚ç‚¹çš„è·ç¦»å¯èƒ½ä¼šå˜çŸ­
                # vec[j] = 0æ²¡æœ‰åŠ å…¥æœ€å°ç”Ÿæˆæ ‘åˆé›†çš„èŠ‚ç‚¹jï¼Œæ›´æ–°jåˆ°æ–°åŠ å…¥èŠ‚ç‚¹min_idxçš„æœ€å°è·ç¦»
                if vec[j] == 0 and graph[min_idx][j] < low_cost[j]:
                    low_cost[j] = graph[min_idx][j]
            # print(vec, low_cost)
    
        return ans
